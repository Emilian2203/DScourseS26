\begin{frame}{Introduction: Why NBA Ticket Pricing?}

\textbf{Problem: Pricing a Differentiated, Perishable Product}

\vspace{0.3cm}
\textbf{Dynamic Pricing in Sports:}
\begin{itemize}
    \item \textbf{Definition:} Demand-based pricing strategy
    \item Widely used in industries such as airlines, hospitality, and sports
    \item Objective: Align prices with consumer willingness to pay (Kopalle et al., 2023)
    \item Common demand signals include:
    \begin{itemize}
        \item Opponent quality and team performance
        \item Injuries to star players
        \item Time remaining until game
    \end{itemize}
\end{itemize}

\vspace{0.2cm}
\textbf{This Paper:} Four research questions across 729,554 seat--game observations at FedEx Forum

\end{frame}

% ── SLIDE 3: METHODS ──
\begin{frame}{Methods: Log-Linear Dynamic Pricing Model}

\textbf{Core idea:} Log-differences cancel seat fixed effects:
$$\Delta^{2wk}_i = \log(P^{2wk}_i) - \log(P^{start}_i)$$

\vspace{0.3cm}
\textbf{Four estimation steps:}

\begin{enumerate}
    \item \textbf{Step 1 — Season-Start Price:} Regress $\log(P^{start})$ on seat level + game controls. Standard errors clustered by game.

    \item \textbf{Step 2 — Start $\to$ Two Weeks Out:} Regress $\Delta^{2wk}$ on game-level demand shifters (N=41, HC3 robust SEs).

    \item \textbf{Step 3 — Two Weeks Out $\to$ Tipoff:} Regress $\Delta^{tip}$ on game-level variables + seat-level interactions.

    \item \textbf{Step 4 — Attendance:} Linear model on 41-game dataset with robust SEs.
\end{enumerate}

\vspace{0.2cm}
\small Software: R (\texttt{fixest}, \texttt{lmtest}, \texttt{sandwich}, \texttt{modelsummary})

\end{frame}

% ── SLIDE 4: FINDINGS ──
\begin{frame}{Findings: Key Results Across All Four Steps}

\begin{columns}

\column{0.5\textwidth}
\textbf{Step 1 — Season-Start}
\begin{itemize}
    \item Seat level explains \textbf{95\%} of price variation (R$^2$ = 0.95)
    \item Courtside: \textbf{+2,733\%} vs Upper Bowl
    \item Each opponent win $\Rightarrow$ \textbf{+0.18\%} price
\end{itemize}

\vspace{0.3cm}
\textbf{Step 3 — To Tipoff}
\begin{itemize}
    \item Effects \textbf{smaller} — market already adjusted
    \item Rivalry premium \textbf{fully absorbed} by 2 weeks out
    \item Lower Bowl: extra \textbf{+8\%} on high demand games
\end{itemize}

\column{0.5\textwidth}
\textbf{Step 2 — To Two Weeks Out}
\begin{itemize}
    \item High demand: \textbf{+22\%} price increase
    \item Rival games: \textbf{+232\%} appreciation
    \item Opp. star injured: \textbf{$-$18\%}
    \item Grizz star injured: \textbf{$-$78\%}
\end{itemize}

\vspace{0.3cm}
\textbf{Step 4 — Attendance}
\begin{itemize}
    \item High demand: \textbf{+2,405 fans}
    \item Opp. star injured: \textbf{$-$1,484 fans}
    \item Grizz star injured: \textbf{$-$794 fans}
    \item Saturday/Sunday: \textbf{+830--898 fans}
\end{itemize}

\end{columns}

\end{frame}

% ── SLIDE 5: CONCLUSIONS ──
\begin{frame}{What Did We Learn? Why Does This Matter?}

\textbf{Key Takeaways:}
\begin{itemize}
  \item \textbf{Seat quality dominates} season-start prices (hedonic pricing)
  \item \textbf{Market is efficient} — most info priced in by two weeks out
  \item \textbf{Star injuries suppress prices} — Grizzlies $-78\%$ vs opponent $-18\%$
  \item \textbf{Premium seats react differently} to marquee games
\end{itemize}

\vspace{0.5em}
\textbf{Why It Matters:}
\begin{itemize}
  \item Helps teams \textbf{optimize revenue} via smarter dynamic pricing
  \item Quantifies the \textbf{economic value} of star players
  \item Shows how secondary markets \textbf{efficiently incorporate information}
\end{itemize}

\vspace{0.5em}
\textbf{Future Work:} multiple seasons, primary market data, IV estimation

\end{frame}
